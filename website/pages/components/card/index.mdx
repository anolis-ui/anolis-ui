# Card

```tsx
import { Card } from "anolis-ui";

interface CardProps extends SystemProps {
  _header?: PseudoProp;
  header?: Renderable;

  _title?: PseudoProp;
  title?: Renderable;

  _tag?: PseudoProp;
  tag?: Renderable;

  _media?: PseudoProp;
  media?: Renderable;

  _body?: PseudoProp;
  body?: Renderable;

  _footer?: PseudoProp;
  footer?: Renderable;
}

type CardVariant = "normal" | "elevated" | "outline";
type CardSize = never;
```
